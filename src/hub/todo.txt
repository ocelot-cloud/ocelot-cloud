Idea for code unzipping:

package main

import (
	"archive/tar"
	"compress/gzip"
	"fmt"
	"io"
	"os"
	"path/filepath"
)

func main() {
	// Path to your tar.gz file
	tarGzFile := "example.tar.gz"
	// Directory to extract the files to
	destDir := "extracted_files"

	if err := unpackTarGz(tarGzFile, destDir); err != nil {
		fmt.Printf("Error unpacking tar.gz file: %v\n", err)
	}
}

func unpackTarGz(src, dest string) error {
	// Open the tar.gz file
	file, err := os.Open(src)
	if err != nil {
		return err
	}
	defer file.Close()

	// Create a gzip reader
	gzReader, err := gzip.NewReader(file)
	if err != nil {
		return err
	}
	defer gzReader.Close()

	// Create a tar reader
	tarReader := tar.NewReader(gzReader)

	// Protection thresholds
	const maxFileSize = 100 * 1024 * 1024 // 100 MB
	const maxFiles = 1000

	// Counters to keep track of total size and number of files
	var totalSize int64
	var fileCount int

	for {
		header, err := tarReader.Next()
		if err == io.EOF {
			break
		}
		if err != nil {
			return err
		}

		// Check file size
		if header.Size > maxFileSize {
			return fmt.Errorf("file too large: %s", header.Name)
		}
		totalSize += header.Size
		fileCount++

		// Check total size and number of files
		if totalSize > maxFileSize {
			return fmt.Errorf("extracted content too large")
		}
		if fileCount > maxFiles {
			return fmt.Errorf("too many files in archive")
		}

		// Construct the file path
		destPath := filepath.Join(dest, header.Name)

		// Handle different file types
		switch header.Typeflag {
		case tar.TypeDir:
			if err := os.MkdirAll(destPath, os.FileMode(header.Mode)); err != nil {
				return err
			}
		case tar.TypeReg:
			if err := extractFile(tarReader, destPath, header.Mode); err != nil {
				return err
			}
		default:
			// Handle other file types if necessary
		}
	}

	return nil
}

func extractFile(reader io.Reader, destPath string, mode int64) error {
	file, err := os.OpenFile(destPath, os.O_CREATE|os.O_WRONLY|os.O_TRUNC, os.FileMode(mode))
	if err != nil {
		return err
	}
	defer file.Close()

	_, err = io.Copy(file, reader)
	return err
}
